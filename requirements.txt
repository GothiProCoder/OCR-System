# INSTALLATION:
#   1. Create/activate the PaddleOCR environment:
#      python -m venv .venv_paddleocr
#      .\.venv_paddleocr\Scripts\Activate.ps1
#
#   2. Install PaddlePaddle first (CPU or GPU):
#      # CPU:
#      pip install paddlepaddle==3.2.1
#      # GPU (CUDA 12.6):
#      pip install paddlepaddle-gpu==3.2.1 -i https://www.paddlepaddle.org.cn/packages/stable/cu126/
#
#   3. Install this file:
#      pip install -r requirements.txt
#
#   4. Install Windows safetensors (if on Windows):
#      pip install https://xly-devops.cdn.bcebos.com/safetensors-nightly/safetensors-0.6.2.dev0-cp38-abi3-win_amd64.whl
